
int count = 0;

ACT generateFibo starring (int number) {
	count++;
	if(number == 1 || number == 2) {
		give 1;
	}

	give generateFibo(number-1)+generateFibo(number-2);
}

ACT main starring () {
	int x;
	int fibo;

	Write to x;

	fibo = generateFibo(x);
	Say "Fibonacci Number "+x+": "+fibo;
	Say count; 

	
}
