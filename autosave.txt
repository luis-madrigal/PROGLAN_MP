
int count = 0;

ACT generateFibo starring (int number) {
	count++;
	if(number == 1 || number == 2) {
		give 1;
	}
	
	give generateFibo(number-1) + generateFibo(number-2);
}

ACT main starring () {
	int x = 0;
	Say "Hi";
	x = 0;
	Say "Hello";
	int fibo;

	[]*Say "Enter a number: ";
	Write to x;

	fibo = generateFibo(x);
	Say "Fibonacci Number "+x+": "+fibo;
	Say count; *[]

	replay (int i = 0; i < 10; i++) {
		replay (int j = 0; j < 10; j++) {
			Say i+j;
		}
	}
}
