
int counter = 0;

ACT solve starring ( int n, string start, string auxilliary, string end) {
	counter++;
	if (n == 1)
		Say start + " -> " + end + ": "+counter;
	else {
		solve(n-1, start, end, auxilliary);
		Say start + " -> " + end;
		solve(n-1, auxilliary, start, end);
	}
}

ACT main starring ()  {
	int x;

	Write to x;

	solve(x, "A", "B", "C");

	Say "final count: "+(counter+1);
}
