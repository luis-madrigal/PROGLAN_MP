
ACT main starring ()  {
	int[] toSort = new int[] {1, 10, 9, 8, 3, 5, 6, 1, 4, 6};
	int len = 10;
	replay ( int i = 0; i < len-1; i++) {
		replay ( int j = 0; j < len-i-1; j++) {
			if ( toSort[j] > toSort[j+1] ) {
				int temp = toSort[j];
				toSort[j] = toSort[j+1];
				toSort[j+1] = temp;
			}
		}
	}
	replay (int i = 0; i < len; i++) {
		Say toSort[i];
	}
}
